** Nix Package
Add ~pkgs/datomic-pro/default.nix~:
#+begin_src nix
{
  fetchzip,
  jre,
  lib,
  makeWrapper,
  stdenv,
  ...
}:
stdenv.mkDerivation (finalAttrs: {
  version = "1.0.7187";
  pname = "datomic-pro";

  src = fetchzip {
    url = "https://datomic-pro-downloads.s3.amazonaws.com/${finalAttrs.version}/datomic-pro-${finalAttrs.version}.zip";
    hash = "sha256-QT+6fozoNQr0Ohic5W+zS2dWEIyso5Sa/OMnwDW1bCA=";
  };

  nativeBuildInputs = [makeWrapper];
  dontConfigure = true;
  dontBuild = true;

  installPhase = ''
    runHook preInstall
    mkdir -p $out/{bin,lib,share}
    cp *transactor*.jar $out/lib/
    mkdir -p $out/share/datomic-pro
    cp -R * $out/share/datomic-pro/

    makeWrapper ${jre}/bin/java $out/bin/datomic-transactor \
      --set-default "JAVA_OPTS" "-XX:+UseG1GC -XX:MaxGCPauseMillis=50" \
      --add-flags "-server" \
      --add-flags "-cp $out/share/datomic-pro/resources:$out/lib/datomic-transactor-pro-${finalAttrs.version}.jar:$out/share/datomic-pro/lib/*" \
      --add-flags "clojure.main --main datomic.launcher"

    runHook postInstall
  '';

  meta = with lib; {
    homepage = "https://www.datomic.com/";
    description = "Datomic Pro transactional database";
    license = licenses.asl20;
    platforms = platforms.all;
  };
})
#+end_src

** Docker: Dockerfile.transactor
#+begin_src dockerfile
FROM eclipse-temurin:21

ARG DATOMIC_VERSION=1.0.7187

RUN apt-get update && apt-get install -y curl libarchive-tools && apt-get clean

WORKDIR /opt/datomic

RUN curl -sL "https://datomic-pro-downloads.s3.amazonaws.com/${DATOMIC_VERSION}/datomic-pro-${DATOMIC_VERSION}.zip" \
    -o /tmp/datomic.zip \
  && bsdtar -xf /tmp/datomic.zip --strip-components=1 -C /opt/datomic \
  && rm /tmp/datomic.zip

EXPOSE 4334 4335 4336

ENTRYPOINT ["bin/transactor"]
CMD ["config/transactor.properties"]
#+end_src

** bits.datomic component changes
Key differences from Datahike:
| Aspect          | Datahike                | Datomic                              |
|-----------------+-------------------------+--------------------------------------|
| Connection arg  | Store map               | URI string                           |
| Get DB value    | ~@conn~                 | ~(d/db conn)~                        |
| Create DB       | ~d/create-database cfg~ | ~(d/create-database uri)~            |
| Delete DB       | ~d/delete-database cfg~ | ~(d/delete-database uri)~            |
